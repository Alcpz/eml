JSON serialization

Energy consumption datasets can be exported as JSON files through emlDataDumpJSON. An emlData JSON object contains a time series of energy data, as well as measurement metadata.

To avoid loss of precision, all values are reported in raw format (that is, in the units native to the method used to obtain the readings), along with rational unit factors to convert them to base SI units.

JSON schema

emlData objects shall conform to the following schema:

{
"$schema": "http://json-schema.org/draft-04/schema#",
"title": "emlData",
"description": "Energy consumption data for a measured time interval",
"definitions": {
"factor": {
"type": "object",
"properties": {
"mult": {
"description": "Factor numerator",
"type": "number",
"minimum": 1
},
"div": {
"description": "Factor denominator",
"type": "number",
"minimum": 1
}
},
"required": ["mult", "div"]
}
},
"type": "object",
"properties": {
"device": {
"description": "EML device identifier",
"type": "string"
},
"elapsed": {
"description": "Elapsed time interval, in time units",
"type": "number",
"minimum": 0
},
"consumed": {
"description": "Consumed energy, in energy units",
"type": "number",
"minimum": 0
},
"time_factor": {
"description": "Factor to convert time units to seconds",
"$ref": "#/definitions/factor"
},
"energy_factor": {
"description": "Factor to convert energy units to seconds",
"$ref": "#/definitions/factor"
},
"power_factor": {
"description": "Factor to convert power units to seconds",
"$ref": "#/definitions/factor"
},
"header": {
"description": "Data fields available",
"type": "array",
"items": {
"type": "string"
},
"uniqueItems": true
},
"data": {
"description": "Data points",
"type": "array"
}
},
"required": [
"device",
"elapsed",
"consumed",
"time_factor",
"energy_factor",
"power_factor",
"header",
"data"
]
}
